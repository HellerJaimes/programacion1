                        bool encontrarMatrizInversa(double A[][10], double inversa[][10], int n) {
                            // Copiar la matriz original a la matriz aumentada
                            double aumentada[10][20];
                            for (int i = 0; i < n; i++) {
                                for (int j = 0; j < n; j++) {
                                    aumentada[i][j] = A[i][j];
                                    aumentada[i][j + n] = (i == j) ? 1 : 0;
                                }
                            }

                            // Hacer operaciones para convertir A en I y guardar resultado en inversa
                            for (int i = 0; i < n; i++) {
                                // Encontrar la fila con el elemento diagonal más grande
                                int filaMax = i;
                                for (int j = i + 1; j < n; j++) {
                                    if (fabs(aumentada[j][i]) > fabs(aumentada[filaMax][i])) {
                                        filaMax = j;
                                    }
                                }

                                // Intercambiar filas si es necesario
                                if (fabs(aumentada[filaMax][i]) < EPSILON) {
                                    // La matriz no es inversible
                                    cout << "La matriz no es inversible." << endl;
                                    return false;
                                }

                                swap(aumentada[i], aumentada[filaMax]);
                                double factor = aumentada[i][i];

                                // Hacer 1 en la diagonal principal
                                for (int j = 0; j < 2 * n; j++) {
                                    aumentada[i][j] /= factor;
                                }

                                // Hacer 0 por encima y por debajo de la diagonal principal
                                for (int j = 0; j < n; j++) {
                                    if (i != j) {
                                        factor = aumentada[j][i];
                                        for (int k = 0; k < 2 * n; k++) {
                                            aumentada[j][k] -= factor * aumentada[i][k];
                                        }
                                    }
                                }
                            }

                            // Extraer la matriz inversa de la parte derecha [I | A^(-1)]
                            for (int i = 0; i < n; i++) {
                                for (int j = 0; j < n; j++) {
                                    inversa[i][j] = aumentada[i][j + n];
                                }
                            }

                            return true;
                        }

                        // Función para calcular el determinante de una matriz
                        double calcularDeterminante(double mat[][10], int n) {
                            double det = 1;
                            for (int i = 0; i < n; i++) {
                                int filaMax = i;
                                for (int j = i + 1; j < n; j++) {
                                    if (fabs(mat[j][i]) > fabs(mat[filaMax][i])) {
                                        filaMax = j;
                                    }
                                }

                                if (fabs(mat[filaMax][i]) < 1e-10) {
                                    return 0; // La matriz es singular (no invertible)
                                }

                                if (filaMax != i) {
                                    swap(mat[i], mat[filaMax]);
                                    det *= -1; // Cambio de signo por intercambio de filas
                                }

                                for (int j = i + 1; j < n; j++) {
                                    double factor = mat[j][i] / mat[i][i];
                                    for (int k = i; k < n; k++) {
                                        mat[j][k] -= factor * mat[i][k];
                                    }
                                }
                            }

                            for (int i = 0; i < n; i++) {
                                det *= mat[i][i];
                            }

                            return det;
                        }

                        // Función para calcular el cofactor de un elemento específico
                        double calcularCofactor(double mat[][10], int n, int fila, int columna) {
                            double cofactorMat[10][10];

                            int cofactorFila = 0;
                            for (int i = 0; i < n; i++) {
                                if (i == fila) continue;

                                int cofactorColumna = 0;
                                for (int j = 0; j < n; j++) {
                                    if (j == columna) continue;

                                    cofactorMat[cofactorFila][cofactorColumna] = mat[i][j];
                                    cofactorColumna++;
                                }

                                cofactorFila++;
                            }

                            // Calcular el determinante del cofactor
                            double detCofactor = calcularDeterminante(cofactorMat, n - 1);

                            // Calcular el cofactor
                            int signo = ((fila + columna) % 2 == 0) ? 1 : -1;
                            double cofactor = signo * detCofactor;

                            return cofactor;
                        }
