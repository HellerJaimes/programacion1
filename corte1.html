                    if (respuesta == 1) {

                        cout << "Por favor Ingrese la longitud del cateto B: ";
                        cin >> catetoB;
                        hipotenusa = sqrt(catetoA * catetoA + catetoB * catetoB);
                        cout << "La hipotenusa es: " << hipotenusa << endl;
                    } 
                    else if (respuesta == 2) {
                        cout << "Por favor Ingrese la longitud del cateto B: ";
                        cin >> catetoB;
                        cout << "Por favor Ingrese la longitud de la hipotenusa: ";
                        cin >> hipotenusa;
                        catetoA = sqrt(hipotenusa * hipotenusa - catetoB * catetoB);
                        cout << "El cateto A es: " << catetoA << endl;
                    } 
                    else if (respuesta == 3) {
                        cout << "Por favor Ingrese la longitud del cateto A: ";
                        cin >> catetoA;
                        cout << "Por favor Ingrese la longitud de la hipotenusa: ";
                        cin >> hipotenusa;
                        catetoB = sqrt(hipotenusa * hipotenusa - catetoA * catetoA);
                        cout << "El cateto B es: " << catetoB << endl;
                    } 
                    else {
                        cout << "Opción no válida." << endl;
                    }
                    
                    return 0;
                }

                    cout << "Ingrese el primer número: ";
                    cin >> num1;
                    cout << "Ingrese el segundo número: ";
                    cin >> num2;
                    cout << "Ingrese el tercer número: ";
                    cin >> num3;
                    
                    int opcion;
                    cout << "Seleccione la opción:" << endl
                    << "1. De mayor a menor" << endl
                    << "2. De menor a mayor" << endl;
                    cin >> opcion;
                    
                    int mayor, medio, menor;
                    
                    mayor = max(num1, max(num2, num3));
                    menor = min(num1, min(num2, num3));
                    medio = num1 + num2 + num3 - mayor - menor;
                    
                    if (opcion == 1) {
                        cout << "De mayor a menor: " << mayor << ", " << medio << ", " << menor
                        << endl;
                    } else if (opcion == 2) {
                        cout << "De menor a mayor: " << menor << ", " << medio << ", " << mayor
                        << endl;
                    } else {
                        cout << "Opción no válida." << endl;
                    }
                    return 0;
                }
