                        #include <iostream>
                        #include <string>

                        int main() {
                            std::cout << "Bienvenido al juego de Ahorcado!\n";

                            std::string palabraSecreta;
                            std::cout << "Ingresa la palabra a adivinar (en minúsculas): ";
                            std::cin >> palabraSecreta;

                            int intentos = 6;
                            std::string palabraAdivinada(palabraSecreta.length(), '_');
                            char letra;

                            while (intentos > 0 && palabraAdivinada != palabraSecreta) {
                                std::cout << "\nPalabra a adivinar: " << palabraAdivinada << std::endl;
                                std::cout << "\nIntentos restantes: " << intentos;
                                std::cout << "\nIngresa una letra: ";
                                std::cin >> letra;

                                bool letraAdivinada = false;

                                for (size_t i = 0; i < palabraSecreta.length(); i++) {
                                    if (palabraSecreta[i] == letra) {
                                        palabraAdivinada[i] = letra;
                                        letraAdivinada = true;
                                    }
                                }

                                if (!letraAdivinada) {
                                    intentos--;
                                    std::cout << "La letra no está en la palabra. Intenta de nuevo.\n";
                                }
                            }

                            if (palabraAdivinada == palabraSecreta) {
                                std::cout << "¡Felicidades! Has adivinado la palabra: " << palabraSecreta;
                            } else {
                                std::cout << "\nAgotaste tus intentos. La palabra secreta era: " << palabraSecreta;
                            }

                            return 0;
                        }
